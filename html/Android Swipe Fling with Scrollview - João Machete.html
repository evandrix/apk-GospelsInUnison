Here is our custom View(the trick):

<br />
public class FlingAndScrollViewer extends ViewGroup {<br />
    private Scroller mScroller;<br />
    private VelocityTracker mVelocityTracker;</p>
<p> private int mScrollX = 0;<br />
    private int mCurrentScreen = 0;</p>
<p> private float mLastMotionX;</p>
<p> private static final String LOG_TAG = "FlingAndScrollViewer";</p>
<p> private static final int SNAP_VELOCITY = 1000;</p>
<p> private final static int TOUCH_STATE_REST = 0;<br />
    private final static int TOUCH_STATE_SCROLLING = 1;</p>
<p> private int mTouchState = TOUCH_STATE_REST;</p>
<p> private int mTouchSlop = 0;</p>
<p> public FlingAndScrollViewer(Context context) {<br />
        super(context);<br />
        mScroller = new Scroller(context);</p>
<p>     mTouchSlop = ViewConfiguration.get(getContext()).getScaledTouchSlop();</p>
<p>     this.setLayoutParams(new ViewGroup.LayoutParams(<br />
                ViewGroup.LayoutParams.WRAP_CONTENT,<br />
                ViewGroup.LayoutParams.FILL_PARENT));<br />
    }</p>
<p> public FlingAndScrollViewer(Context context, AttributeSet attrs) {<br />
        super(context, attrs);<br />
        TypedArray a = getContext().obtainStyledAttributes(attrs,<br />
                R.styleable.FlingAndScrollViewer);<br />
        mCurrentScreen = a.getInteger(<br />
                R.styleable.FlingAndScrollViewer_default_screen, 0);</p>
<p>     mScroller = new Scroller(context);</p>
<p>     mTouchSlop = ViewConfiguration.get(getContext()).getScaledTouchSlop();</p>
<p>     this.setLayoutParams(new ViewGroup.LayoutParams(<br />
                ViewGroup.LayoutParams.WRAP_CONTENT,<br />
                ViewGroup.LayoutParams.FILL_PARENT));</p>
<p> }</p>
<p> public void setInitialPosition(int initialPosition){<br />
        mCurrentScreen = initialPosition;<br />
    }</p>
<p> @Override<br />
    public boolean onInterceptTouchEvent(MotionEvent ev) {<br />
        /*<br />
         * This method JUST determines whether we want to intercept the motion.<br />
         * If we return true, onTouchEvent will be called and we do the actual<br />
         * scrolling there.<br />
         */</p>
<p>     /*<br />
         * Shortcut the most recurring case: the user is in the dragging state<br />
         * and he is moving his finger. We want to intercept this motion.<br />
         */<br />
        final int action = ev.getAction();<br />
        if ((action == MotionEvent.ACTION_MOVE)<br />
                &#038;& (mTouchState != TOUCH_STATE_REST)) {<br />
            return true;<br />
        }</p>
<p>     final float x = ev.getX();</p>
<p>     switch (action) {<br />
        case MotionEvent.ACTION_MOVE:<br />
            /*<br />
             * mIsBeingDragged == false, otherwise the shortcut would have<br />
             * caught it. Check whether the user has moved far enough from his<br />
             * original down touch.<br />
             */</p>
<p>         /*<br />
             * Locally do absolute value. mLastMotionX is set to the y value of<br />
             * the down event.<br />
             */<br />
            final int xDiff = (int) Math.abs(x - mLastMotionX);</p>
<p>         boolean xMoved = xDiff > mTouchSlop;</p>
<p>         if (xMoved) {<br />
                // Scroll if the user moved far enough along the X axis<br />
                mTouchState = TOUCH_STATE_SCROLLING;<br />
            }<br />
            break;</p>
<p>     case MotionEvent.ACTION_DOWN:<br />
            // Remember location of down touch<br />
            mLastMotionX = x;</p>
<p>         /*<br />
             * If being flinged and user touches the screen, initiate drag;<br />
             * otherwise don't. mScroller.isFinished should be false when being<br />
             * flinged.<br />
             */<br />
            mTouchState = mScroller.isFinished() ? TOUCH_STATE_REST<br />
                    : TOUCH_STATE_SCROLLING;<br />
            break;</p>
<p>     case MotionEvent.ACTION_CANCEL:<br />
        case MotionEvent.ACTION_UP:<br />
            // Release the drag<br />
            mTouchState = TOUCH_STATE_REST;<br />
            break;<br />
        }</p>
<p>     /*<br />
         * The only time we want to intercept motion events is if we are in the<br />
         * drag mode.<br />
         */<br />
        return mTouchState != TOUCH_STATE_REST;<br />
    }</p>
<p> @Override<br />
    public boolean onTouchEvent(MotionEvent event) {</p>
<p>     if (mVelocityTracker == null) {<br />
            mVelocityTracker = VelocityTracker.obtain();<br />
        }<br />
        mVelocityTracker.addMovement(event);</p>
<p>     final int action = event.getAction();<br />
        final float x = event.getX();</p>
<p>     switch (action) {<br />
        case MotionEvent.ACTION_DOWN:<br />
            Log.i(LOG_TAG, "event : down");<br />
            /*<br />
             * If being flinged and user touches, stop the fling. isFinished<br />
             * will be false if being flinged.<br />
             */<br />
            if (!mScroller.isFinished()) {<br />
                mScroller.abortAnimation();<br />
            }</p>
<p>         // Remember where the motion event started<br />
            mLastMotionX = x;<br />
            break;<br />
        case MotionEvent.ACTION_MOVE:<br />
            // Log.i(LOG_TAG,"event : move");<br />
            // if (mTouchState == TOUCH_STATE_SCROLLING) {<br />
            // Scroll to follow the motion event<br />
            final int deltaX = (int) (mLastMotionX - x);<br />
            mLastMotionX = x;</p>
<p>         // Log.i(LOG_TAG, "event : move, deltaX " + deltaX + ", mScrollX " +<br />
            // mScrollX);</p>
<p>         if (deltaX < 0) {<br />
                if (mScrollX > 0) {<br />
                    scrollBy(Math.max(-mScrollX, deltaX), 0);<br />
                }<br />
            } else if (deltaX > 0) {<br />
                final int availableToScroll = getChildAt(getChildCount() - 1)<br />
                        .getRight() - mScrollX - getWidth();<br />
                if (availableToScroll > 0) {<br />
                    scrollBy(Math.min(availableToScroll, deltaX), 0);<br />
                }<br />
            }<br />
            // }<br />
            break;<br />
        case MotionEvent.ACTION_UP:<br />
            Log.i(LOG_TAG, "event : up");<br />
            // if (mTouchState == TOUCH_STATE_SCROLLING) {<br />
            final VelocityTracker velocityTracker = mVelocityTracker;<br />
            velocityTracker.computeCurrentVelocity(1000);<br />
            int velocityX = (int) velocityTracker.getXVelocity();</p>
<p>         if (velocityX > SNAP_VELOCITY &#038;& mCurrentScreen > 0) {<br />
                // Fling hard enough to move left<br />
                snapToScreen(mCurrentScreen - 1);<br />
            } else if (velocityX < -SNAP_VELOCITY<br />
                    &#038;& mCurrentScreen < getChildCount() - 1) {<br />
                // Fling hard enough to move right<br />
                snapToScreen(mCurrentScreen + 1);<br />
            } else {<br />
                snapToDestination();<br />
            }</p>
<p>         if (mVelocityTracker != null) {<br />
                mVelocityTracker.recycle();<br />
                mVelocityTracker = null;<br />
            }<br />
            // }<br />
            mTouchState = TOUCH_STATE_REST;<br />
            break;<br />
        case MotionEvent.ACTION_CANCEL:<br />
            Log.i(LOG_TAG, "event : cancel");<br />
            mTouchState = TOUCH_STATE_REST;<br />
        }<br />
        mScrollX = this.getScrollX();</p>
<p>     return true;<br />
    }</p>
<p> private void snapToDestination() {<br />
        final int screenWidth = getWidth();<br />
        final int whichScreen = (mScrollX + (screenWidth / 2)) / screenWidth;<br />
        Log.i(LOG_TAG, "from des");<br />
        snapToScreen(whichScreen);<br />
    }</p>
<p> public void snapToScreen(int whichScreen) {<br />
        Log.i(LOG_TAG, "snap To Screen " + whichScreen);<br />
        mCurrentScreen = whichScreen;<br />
        final int newX = whichScreen * getWidth();<br />
        final int delta = newX - mScrollX;<br />
        mScroller.startScroll(mScrollX, 0, delta, 0, Math.abs(delta) * 2);<br />
        invalidate();<br />
    }</p>
<p> public void setToScreen(int whichScreen) {<br />
        Log.i(LOG_TAG, "set To Screen " + whichScreen);<br />
        mCurrentScreen = whichScreen;<br />
        final int newX = whichScreen * getWidth();<br />
        mScroller.startScroll(newX, 0, 0, 0, 10);<br />
        invalidate();<br />
    }</p>
<p> @Override<br />
    protected void onLayout(boolean changed, int l, int t, int r, int b) {<br />
        int childLeft = 0;</p>
<p>     final int count = getChildCount();<br />
        for (int i = 0; i < count; i++) {<br />
            final View child = getChildAt(i);<br />
            if (child.getVisibility() != View.GONE) {<br />
                final int childWidth = child.getMeasuredWidth();<br />
                child.layout(childLeft, 0, childLeft + childWidth,<br />
                        child.getMeasuredHeight());<br />
                childLeft += childWidth;<br />
            }<br />
        }</p>
<p> }</p>
<p> @Override<br />
    protected void onMeasure(int widthMeasureSpec, int heightMeasureSpec) {<br />
        super.onMeasure(widthMeasureSpec, heightMeasureSpec);</p>
<p>     final int width = MeasureSpec.getSize(widthMeasureSpec);<br />
        final int widthMode = MeasureSpec.getMode(widthMeasureSpec);<br />
        if (widthMode != MeasureSpec.EXACTLY) {<br />
            throw new IllegalStateException("error mode.");<br />
        }</p>
<p>     final int heightMode = MeasureSpec.getMode(heightMeasureSpec);<br />
        if (heightMode != MeasureSpec.EXACTLY) {<br />
            throw new IllegalStateException("error mode.");<br />
        }</p>
<p>     // The children are given the same width and height as the workspace<br />
        final int count = getChildCount();<br />
        for (int i = 0; i < count; i++) {<br />
            getChildAt(i).measure(widthMeasureSpec, heightMeasureSpec);<br />
        }<br />
        Log.i(LOG_TAG, "moving to screen " + mCurrentScreen);<br />
        scrollTo(mCurrentScreen * width, 0);<br />
    }</p>
<p> @Override<br />
    public void computeScroll() {<br />
        if (mScroller.computeScrollOffset()) {<br />
            mScrollX = mScroller.getCurrX();<br />
            scrollTo(mScrollX, 0);<br />
            postInvalidate();<br />
        }<br />
    }</p>
<p> /**<br />
     * Return the parceable instance to be saved<br />
     */<br />
    @Override<br />
    protected Parcelable onSaveInstanceState() {<br />
        final SavedState state = new SavedState(super.onSaveInstanceState());<br />
        state.currentScreen = mCurrentScreen;<br />
        return state;<br />
    }</p>
<p> /**<br />
     * Restore the previous saved current screen<br />
     */<br />
    @Override<br />
    protected void onRestoreInstanceState(Parcelable state) {<br />
        SavedState savedState = (SavedState) state;<br />
        super.onRestoreInstanceState(savedState.getSuperState());<br />
        if (savedState.currentScreen != -1) {<br />
            mCurrentScreen = savedState.currentScreen;<br />
        }<br />
    }</p>
<p> // ========================= INNER CLASSES ==============================</p>
<p> public interface onViewChangedEvent {<br />
        void onViewChange(int currentViewIndex);<br />
    }</p>
<p> /**<br />
     * A SavedState which save and load the current screen<br />
     */<br />
    public static class SavedState extends BaseSavedState {<br />
        int currentScreen = -1;</p>
<p>     /**<br />
         * Internal constructor<br />
         *<br />
         * @param superState<br />
         */<br />
        SavedState(Parcelable superState) {<br />
            super(superState);<br />
        }</p>
<p>     /**<br />
         * Private constructor<br />
         *<br />
         * @param in<br />
         */<br />
        private SavedState(Parcel in) {<br />
            super(in);<br />
            currentScreen = in.readInt();<br />
        }</p>
<p>     /**<br />
         * Save the current screen<br />
         */<br />
        @Override<br />
        public void writeToParcel(Parcel out, int flags) {<br />
            super.writeToParcel(out, flags);<br />
            out.writeInt(currentScreen);<br />
        }</p>
<p>     /**<br />
         * Return a Parcelable creator<br />
         */<br />
        public static final Parcelable.Creator<SavedState> CREATOR = new Parcelable.Creator<SavedState>() {<br />
            public SavedState createFromParcel(Parcel in) {<br />
                return new SavedState(in);<br />
            }</p>
<p>         public SavedState[] newArray(int size) {<br />
                return new SavedState[size];<br />
            }<br />
        };<br />
    }<br />
}<br />

To finalize this, create a attrs.xml inside yours res/values:

<?xml version="1.0" encoding="utf-8"?><br />
<resources><br />
    <declare-styleable name="FlingAndScrollViewer"><br />
        <attr name="default_screen" format="integer"/><br />
    </declare-styleable><br />
</resources>
